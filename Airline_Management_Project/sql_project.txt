-- Passenger table --
create table passenger(
    passenger_id int primary key auto_increment,
    pass_name varchar(50) not null,
    gender enum('M','F','Other') not null,
    date_of_birth date not null,
    id_proof_type enum('passport','adhar','voter_id') not null,
   id_proof_number varchar(30) not null unique,
    mob_no bigint unique,
    email varchar(100) unique

);



create table aircraft(
    aircraft_id int primary key auto_increment,
    reg_no varchar(20) not null unique,
    model varchar(50) not null,
    total_seats int not null check(total_seats>0)
);


create table flight
(flight_id int primary key auto_increment,
flight_no varchar(15) not null unique,
source_airport int not null,
destination_airport int not null,
arrival_time datetime not null,
departure_time datetime not null,
aircraft_id int not null,
check(source_airport <> destination_airport),
foreign key(aircraft_id) references aircraft(aircraft_id),
foreign key(source_airport) references airport(airport_id),
foreign key(destination_airport) references airport(airport_id)

);


create table fare
(fare_id int primary key auto_increment,
seat_class enum('economy','premium_economy','business','first') not null,
price decimal(10,2) not null,
flight_id int not null,
foreign key(flight_id) references flight(flight_id));



create table booking
(booking_id int primary key auto_increment,
booking_date datetime not null,
booking_status enum('confirmed','cancelled','pending') not null,
passenger_id int not null,
flight_id int not null,
foreign key(passenger_id) references passenger(passenger_id),
foreign key(flight_id) references flight(flight_id));



create table payment
(payment_id int primary key auto_increment,
amount decimal(10,2) not null ,
payment_status enum('pending','success','failed','refunded'),
payment_date datetime not null,
booking_id int not null,
foreign key(booking_id) references booking(booking_id));


create table crew
(crew_id int primary key auto_increment,
crew_name varchar(50) not null,
role varchar(50) not null
);



create table itinerary
(itinerary_id int primary key auto_increment,
passenger_id int not null,
foreign key(passenger_id) references passenger(passenger_id));


create table itinerary_flight
(itinerary_id int,
flight_id int,
flight_sequence int not null,
foreign key(itinerary_id) references itinerary(itinerary_id),
foreign key(flight_id) references flight(flight_id));


create table crew_assignment
(crew_id int,
flight_id int,
assignment_date datetime,
foreign key(crew_id) references crew(crew_id),
foreign key(flight_id) references flight(flight_id));


create table aircraft_assignment
(aircraft_id int,
flight_id int,
assignment_date datetime,
foreign key(aircraft_id) references aircraft(aircraft_id),
foreign key(flight_id) references flight(flight_id));



CREATE TABLE airport (
    airport_id INT PRIMARY KEY AUTO_INCREMENT,
    city_name VARCHAR(50) UNIQUE,
    latitude  DECIMAL(9,6) NOT NULL,
    longitude DECIMAL(9,6) NOT NULL
);

CREATE TRIGGER trg_manage_seats_on_booking_update
AFTER UPDATE ON booking
FOR EACH ROW
BEGIN
    DECLARE curr_seats INT;

    -- Case 1: Booking CONFIRMED → reduce seat
    IF NEW.booking_status = 'confirmed'
       AND OLD.booking_status <> 'confirmed' THEN

        UPDATE aircraft a
        JOIN flight f ON f.aircraft_id = a.aircraft_id
        SET a.total_seats = a.total_seats - 1
        WHERE f.flight_id = NEW.flight_id;

    END IF;

    -- Case 2: Booking CANCELLED or PENDING after CONFIRM → increase seat
    IF OLD.booking_status = 'confirmed'
       AND NEW.booking_status IN ('cancelled','pending') THEN

        UPDATE aircraft a
        JOIN flight f ON f.aircraft_id = a.aircraft_id
        SET a.total_seats = a.total_seats + 1
        WHERE f.flight_id = NEW.flight_id;

    END IF;

    -- Safety check: no negative seats
    SELECT a.total_seats
    INTO curr_seats
    FROM aircraft a
    JOIN flight f ON f.aircraft_id = a.aircraft_id
    WHERE f.flight_id = NEW.flight_id;

    IF curr_seats < 0 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'No seats available';
    END IF;

END$$

DELIMITER ;


 CREATE TRIGGER calculate_fare_before_insert
     BEFORE INSERT ON fare
     FOR EACH ROW
     BEGIN
     DECLARE dist DECIMAL(10,2);
   
    -- Calculate distance using Haversine formula
     SELECT 6371 * ACOS(
     
      COS(RADIANS(a1.latitude)) *
      COS(RADIANS(a2.latitude)) *
      COS(RADIANS(a2.longitude) - RADIANS(a1.longitude)) +
      SIN(RADIANS(a1.latitude)) *
      SIN(RADIANS(a2.latitude))
      ) INTO dist
      FROM flight fl
      JOIN airport a1 ON fl.source_airport = a1.airport_id
      JOIN airport a2 ON fl.destination_airport = a2.airport_id
      WHERE fl.flight_id = NEW.flight_id;
   
       -- Multiply by seat class multiplier
       SET NEW.price = ROUND(dist * 5 *
          CASE
              WHEN NEW.seat_class='economy' THEN 1.0
                 WHEN NEW.seat_class='premium_economy' THEN 1.3
               WHEN NEW.seat_class='business' THEN 1.8
              WHEN NEW.seat_class='first' THEN 2.5
          END
       ,2);
END;
